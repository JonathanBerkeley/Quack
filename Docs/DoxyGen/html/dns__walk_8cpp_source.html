44DnsEntries GetCachedDNSData(const bool cached_load) {
45 std::vector<DnsEntry> entries{};
46
47 DnsTablePtr table_entry = nullptr;
48
49 HMODULE dns_lib = cached_load
50 ? CachedLoadLibrary(L"DNSAPI.dll")
51 : LoadLibrary(L"DNSAPI.dll");
52
53 if (dns_lib == nullptr)
54 return std::nullopt;
55
56 // Obtain pointer to undocumented DnsGetCacheDataTable function in DNSAPI
57 FARPROC DnsGetCacheDataTableFunc = GetProcAddress(dns_lib, "DnsGetCacheDataTable");
58 const auto DnsGetCacheDataTable = reinterpret_cast<DnsTableCachePtr>(DnsGetCacheDataTableFunc);
59
60 // Obtain pointer to DnsFree function in DNSAPI
61 FARPROC DnsFreeFunc = GetProcAddress(dns_lib, "DnsFree");
62 const auto DnsFree = reinterpret_cast<DnsFreePtr>(DnsFreeFunc);
63
64 // Gets pointer to beginning of table
65 DnsGetCacheDataTable(&table_entry);
66
67 if (table_entry == nullptr) {
68 DnsFree(table_entry, DnsFreeFlat);
69 return std::nullopt;
70 }
71
72 // Enumerate the cache
73 while (table_entry) {
74 // Store copy of entry that is found
75 DnsEntry this_entry;
76 this_entry.name = std::wstring{ table_entry->name };
77 this_entry.type = table_entry->type;
78 entries.emplace_back(this_entry);
79
80 // Cleanup
81 DnsFree(table_entry->name, DnsFreeFlat);
82 PVOID previous_p_entry = table_entry;
83
84 // Go to next entry
85 table_entry = table_entry->next;
86
87 DnsFree(previous_p_entry, DnsFreeFlat);
88 }
89
90 // todo: Verify no leaks possible
91
92 // Cleanup
93 if (!cached_load) {
94 FreeLibrary(dns_lib);
95 }
96
97 return IfNotEmpty(entries);
98}
