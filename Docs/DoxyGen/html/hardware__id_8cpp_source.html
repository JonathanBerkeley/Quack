30 // Retrieve size needed for table

32 'RSMB',
33 0,
34 nullptr,
35 0
36 );
37
38 // Allocate room for the table
39 const auto smbios_data = static_cast<smbios::RawSMBIOSData*>(HeapAlloc(GetProcessHeap(), 0, smbios_size));
40 if (not smbios_data)
41 return {};
42
43 // Retrieve SMBIOS table
44 const auto bytes_written = GetSystemFirmwareTable('RSMB', 0, smbios_data, smbios_size);
45 if (not bytes_written)
46 return {};
47
48 const auto smbios_table = smbios_data->SMBIOSTableData;
49
50 // Create HWID
51 std::stringstream hwid;
52 hwid << static_cast<int>(smbios_table[0x4]) << static_cast<int>(smbios_table[0x5]) << static_cast<int>(smbios_table[0x8]);
53
54 if constexpr (cfg::Features::GpuID)
55 hwid << GpuID().value_or("");
56
57 if constexpr (cfg::Features::DiskID)
58 hwid << DiskID().value_or(0);
59
60 if constexpr (cfg::Features::CpuID)
61 hwid << static_cast<int>(GetCpuID());
62
63 // Cleanup
64 HeapFree(GetProcessHeap(), 0, smbios_data);
