153bool is_proc_running(const wchar_t* proc_name) {
154 bool is_running = false;
155 PROCESSENTRY32 process_entry{};
156 process_entry.dwSize = sizeof(PROCESSENTRY32);
157
158 HANDLE proc_snap = CreateToolhelp32Snapshot(TH32CS_SNAPPROCESS, NULL);
159
160 if (Process32First(proc_snap, &process_entry))
161 while (Process32Next(proc_snap, &process_entry))
162 if (!_wcsicmp(process_entry.szExeFile, proc_name))
163 is_running = true;
164
165 CloseHandle(proc_snap);
166 return is_running;
167}
