17std::unique_ptr<BYTE[]> parse_dll(const char* dll_name) {
18 if (!GetFileAttributesA(dll_name)) {
19 log_write("(ERROR_MM) File doesn't exist");
20 return nullptr;
21 }
22
23 std::ifstream file(dll_name, std::ios::binary | std::ios::ate);
24 if (file.fail()) {
25 log_write("(ERROR_MM) Opening file failed: " + std::to_string(static_cast<DWORD>(file.rdstate())));
26 return nullptr;
27 }
28
29 const auto file_size = file.tellg();
30 if (file_size < 0x1000) {
31 log_write("(ERROR_MM) File-size invalid");
32 return nullptr;
33 }
34
35 auto src_data = std::make_unique<BYTE[]>(static_cast<size_t>(file_size));
36 if (!src_data) {
37 log_write("(ERROR_MM) Memory allocating failed");
38 return nullptr;
39 }
40
41 file.seekg(0, std::ios::beg);
42 file.read(reinterpret_cast<char*>(src_data.get()), file_size);
43 return src_data;
44}

53HANDLE manual_map(HANDLE proc_handle, const char* dll_name) {
54
55 const auto src_data = parse_dll(dll_name);
56
57 if (reinterpret_cast<IMAGE_DOS_HEADER*>(src_data.get())->e_magic != 0x5A4D) {
58 log_write("(ERROR_MM) Wrong file type");
59 return nullptr;
60 }
61
62 auto old_nt_header = reinterpret_cast<IMAGE_NT_HEADERS*>(
63 src_data.get() + reinterpret_cast<IMAGE_DOS_HEADER*>(src_data.get())->e_lfanew);
64
65 const IMAGE_OPTIONAL_HEADER* old_opt_header = &old_nt_header->OptionalHeader;
66 const IMAGE_FILE_HEADER* old_file_header = &old_nt_header->FileHeader;
67
68#ifdef _WIN64
69 if (old_file_header->Machine != IMAGE_FILE_MACHINE_AMD64) {
70 log_write("(ERROR_MM) Unsupported platform");
71 return nullptr;
72 }
73#else
74 if (old_file_header->Machine != IMAGE_FILE_MACHINE_I386) {
75 log_write("(ERROR_MM) Unsupported platform");
76 return nullptr;
77 }
78#endif
79
80 auto target_base = static_cast<BYTE*>(
81 VirtualAllocEx(
82 proc_handle,
83 reinterpret_cast<void*>(old_opt_header->ImageBase),
84 old_opt_header->SizeOfImage,
85 MEM_COMMIT | MEM_RESERVE,
86 PAGE_EXECUTE_READWRITE
87 ));
88
89 if (!target_base) {
90 target_base = static_cast<BYTE*>(VirtualAllocEx(proc_handle, nullptr, old_opt_header->SizeOfImage,
91 MEM_COMMIT | MEM_RESERVE, PAGE_EXECUTE_READWRITE));
92
93 if (!target_base) {
94 log_write("(ERROR_MM) Memory allocation failed: " + std::to_string(GetLastError()));
95 return nullptr;
96 }
97 }
98
99 MappingData mapping_data{};
100 mapping_data.LoadLibraryA_ptr = LoadLibraryA;
101 mapping_data.GetProcAddress_ptr = reinterpret_cast<GetProcAddress_fn>(GetProcAddress); // NOLINT(clang-diagnostic-cast-function-type)
102
103 auto* section_header = IMAGE_FIRST_SECTION(old_nt_header);
104 for (UINT i = 0; i != old_file_header->NumberOfSections; ++i, ++section_header) {
105 if (section_header->SizeOfRawData) {
106 if (!WriteProcessMemory(proc_handle, target_base + section_header->VirtualAddress,
107 src_data.get() + section_header->PointerToRawData, section_header->SizeOfRawData, nullptr)) {
108 log_write("(ERROR_MM) Could not map sections: " + std::to_string(GetLastError()));
109 VirtualFreeEx(proc_handle, target_base, 0, MEM_RELEASE);
110 return nullptr;
111 }
112 }
113 }
114 memcpy(src_data.get(), &mapping_data, sizeof(mapping_data));
115 WriteProcessMemory(proc_handle, target_base, src_data.get(), 0x1000, nullptr);
116
117 void* alloc_memory = VirtualAllocEx(
118 proc_handle,
119 nullptr,
120 0x1000,
121 MEM_COMMIT | MEM_RESERVE,
122 PAGE_EXECUTE_READWRITE);
123 if (!alloc_memory) {
124 log_write("(ERROR_MM) Shell-code memory allocation failed: " + std::to_string(GetLastError()));
125 VirtualFreeEx(proc_handle, target_base, 0, MEM_RELEASE);
126 return nullptr;
127 }
128
129 WriteProcessMemory(proc_handle, alloc_memory, shell_code, 0x1000, nullptr);
130
131 HANDLE new_thread = CreateRemoteThread(
132 proc_handle,
133 nullptr,
134 0,
135 reinterpret_cast<LPTHREAD_START_ROUTINE>(alloc_memory),
136 target_base,
137 0,
138 nullptr
139 );
140
141 // If CreateRemoteThread causes the process to crash, this won't know
142 if (!new_thread) {
143 log_write("(ERROR_MM) CreateRemoteThread failed: " + std::to_string(GetLastError()));
144 VirtualFreeEx(proc_handle, target_base, 0, MEM_RELEASE);
145 VirtualFreeEx(proc_handle, alloc_memory, 0, MEM_RELEASE);
146 return nullptr;
147 }
148
149 HINSTANCE hCheck = nullptr;
150 while (!hCheck) {
151
152 // Prevent infinite loop on target process death
153 DWORD exit_code{};
154 if (GetExitCodeProcess(proc_handle, &exit_code); exit_code != STILL_ACTIVE) {
155 log_write("(ERROR_MM) Target process died while attempting to manual map to it");
156 return nullptr;
157 }
158
159 MappingData data_checked{};
160 ReadProcessMemory(proc_handle, target_base, &data_checked, sizeof(data_checked), nullptr);
161 hCheck = data_checked.Module;
162 Sleep(10);
163 }
164
165 VirtualFreeEx(proc_handle, alloc_memory, 0, MEM_RELEASE);
166
167 return new_thread;
168}

183void WINAPI shell_code(MappingData* mapping_data) {
184
185 if (!mapping_data)
186 return;
187
188 const auto base = reinterpret_cast<BYTE*>(mapping_data);
189
190 auto* e_lfanew = base + reinterpret_cast<IMAGE_DOS_HEADER*>(mapping_data)->e_lfanew;
191 const auto* optional_headers = &reinterpret_cast<IMAGE_NT_HEADERS*>(e_lfanew)->OptionalHeader;
192
193 const auto load_library_a = mapping_data->LoadLibraryA_ptr;
194 const auto get_proc_address = mapping_data->GetProcAddress_ptr;
195 const auto dll_main = reinterpret_cast<DllEntryPoint_fn>(base + optional_headers->AddressOfEntryPoint);
196
197 if (BYTE* location_delta = base - optional_headers->ImageBase) {
198
199 // Check if possible to relocate data
200 if (!optional_headers->DataDirectory[IMAGE_DIRECTORY_ENTRY_BASERELOC].Size)
201 return;
202
203 const auto base_reloc_vaddr = optional_headers->DataDirectory[IMAGE_DIRECTORY_ENTRY_BASERELOC].VirtualAddress;
204 auto* base_relocation = reinterpret_cast<IMAGE_BASE_RELOCATION*>(base + base_reloc_vaddr);
205
206 while (base_relocation->VirtualAddress) {
207
208 const UINT entry_count = (base_relocation->SizeOfBlock - sizeof(IMAGE_BASE_RELOCATION)) / sizeof(WORD);
209 auto relative_info = reinterpret_cast<WORD*>(base_relocation + 1);
210
211 for (UINT i = 0u; i != entry_count; ++i, ++relative_info) {
212
213 if (RELOC_FLAG(*relative_info)) {
214 const auto reloc_addr = base + base_relocation->VirtualAddress + (*relative_info & 0xFFF);
215 const auto patch = reinterpret_cast<UINT_PTR*>(reloc_addr);
216
217 *patch += reinterpret_cast<UINT_PTR>(location_delta);
218 }
219
220 }
221 base_relocation = reinterpret_cast<IMAGE_BASE_RELOCATION*>(
222 reinterpret_cast<BYTE*>(base_relocation) + base_relocation->SizeOfBlock);
223 }
224 }
225
226 // Check for import directory
227 if (const auto [VirtualAddress, Size]
228 = optional_headers->DataDirectory[IMAGE_DIRECTORY_ENTRY_IMPORT];
229 Size) {
230
231 auto* import_descriptor = reinterpret_cast<IMAGE_IMPORT_DESCRIPTOR*>(base + VirtualAddress);
232
233 while (import_descriptor->Name) {
234
235 const char* sz_mod = reinterpret_cast<char*>(base + import_descriptor->Name);
236
237 // ReSharper disable once CppLocalVariableMayBeConst
238 HINSTANCE dll = load_library_a(sz_mod);
239
240 auto thunk_ref = reinterpret_cast<ULONG_PTR*>(base + import_descriptor->OriginalFirstThunk);
241 auto* func_ref = reinterpret_cast<ULONG_PTR*>(base + import_descriptor->FirstThunk);
242
243 if (!thunk_ref)
244 thunk_ref = func_ref;
245
246 for (; *thunk_ref; ++thunk_ref, ++func_ref) {
247 if (IMAGE_SNAP_BY_ORDINAL(*thunk_ref)) {
248 *func_ref = get_proc_address(dll, reinterpret_cast<char*>(*thunk_ref & 0xFFFF));
249 }
250 else {
251 const auto* import_by_name = reinterpret_cast<IMAGE_IMPORT_BY_NAME*>(base + *thunk_ref);
252 *func_ref = get_proc_address(dll, import_by_name->Name);
253 }
254 }
255 ++import_descriptor;
256 }
257 }
258
259 // Check for thread local storage
260 // https://docs.microsoft.com/en-us/windows/win32/debug/pe-format#the-tls-directory
261 if (const auto [VirtualAddress, Size]
262 = optional_headers->DataDirectory[IMAGE_DIRECTORY_ENTRY_TLS];
263 Size) {
264
265 const auto* tls = reinterpret_cast<IMAGE_TLS_DIRECTORY*>(base + VirtualAddress);
266
267 for (auto* tls_callback = reinterpret_cast<PIMAGE_TLS_CALLBACK*>(tls->AddressOfCallBacks); tls_callback && *tls_callback; ++tls_callback)
268 (*tls_callback)(base, DLL_PROCESS_ATTACH, nullptr);
269
270 }
271 dll_main(base, DLL_PROCESS_ATTACH, nullptr);
272 mapping_data->Module = reinterpret_cast<HINSTANCE>(base);
273}
