28bool PointerIsValid(const void* ptr) {
29 MEMORY_BASIC_INFORMATION mbi{};
30
31 if (VirtualQuery(ptr, &mbi, sizeof mbi)) {
32 constexpr auto mask = PAGE_READONLY | PAGE_READWRITE | PAGE_WRITECOPY | PAGE_EXECUTE_READ | PAGE_EXECUTE_READWRITE | PAGE_EXECUTE_WRITECOPY;
33 bool protect = mbi.Protect & mask;
34
35 if (mbi.Protect & (PAGE_GUARD | PAGE_NOACCESS))
36 protect = false;
37
38 return protect;
39 }
40
41 return false;
42}

72bool VerifyModule(const LPCWSTR source_file) {
73 WINTRUST_FILE_INFO file_data{};
74 file_data.cbStruct = sizeof(WINTRUST_FILE_INFO);
75 file_data.pcwszFilePath = source_file;
76
77 GUID policy_guid = WINTRUST_ACTION_GENERIC_VERIFY_V2;
78 WINTRUST_DATA win_trust_data{
79 .cbStruct = sizeof(win_trust_data),
80 .pPolicyCallbackData = nullptr, // Use default code signing EKU
81 .pSIPClientData = nullptr, // Not applicable
82 .dwUIChoice = WTD_UI_NONE, // Disable WVT UI
83 .fdwRevocationChecks = WTD_REVOKE_NONE, // No revocation checking.
84 .dwUnionChoice = WTD_CHOICE_FILE, // Verify an embedded signature on a file
85 .pFile = &file_data, // Setup pointer to file data
86 .dwStateAction = WTD_STATEACTION_VERIFY, // Verify action
87 .hWVTStateData = nullptr, // Verification sets this value
88 .dwUIContext = 0 // Not applicable with no UI
89 };
90
91 bool verified = false;
92
93 // WinVerifyTrust verifies signatures as specified by the GUID and win_trust_data
94 if (WinVerifyTrust(nullptr, &policy_guid, &win_trust_data) == ERROR_SUCCESS) {
95 verified = true;
96 }
97
98 // Any hWVTStateData must be released by a call with close.
99 win_trust_data.dwStateAction = WTD_STATEACTION_CLOSE;
100 return verified;
101}

111std::vector<unsigned long> PatternToByte(const std::string& signature) {
112 // Pattern of bytes to be returned
113 std::vector<unsigned long> bytes{};
114
115 // Pointer to the start of the signature
116 const auto c_signature = const_cast<char*>(signature.c_str());
117
118 // Start and end of pattern
119 const auto start = c_signature;
120 const auto end = c_signature + signature.length();
121
122 // Loop over pattern
123 for (char* byte = start; byte < end; ++byte) {
124
125 // Check if signature has wildcard at this position
126 if (*byte == '*') {
127 ++byte;
128 if (*byte == '*')
129 ++byte;
130
131 // Flag value pushed back
132 bytes.push_back(0xDEADBEEF);
133 }
134 else {
135 // Store as hex
136 bytes.push_back(std::strtoul(byte, &byte, 16));
137 }
138 }
139
140 return bytes;
141}

153std::uint8_t* PatternScan(const HMODULE module, const std::string& signature) {
154 const auto dos_header = reinterpret_cast<PIMAGE_DOS_HEADER>(module);
155 const auto e_lfanew = reinterpret_cast<std::uint8_t*>(module) + dos_header->e_lfanew;
156 const auto nt_headers = reinterpret_cast<PIMAGE_NT_HEADERS>(e_lfanew);
157
158 const auto size_of_image = nt_headers->OptionalHeader.SizeOfImage;
159 const auto pattern_bytes = PatternToByte(signature);
160 const auto scan_bytes = reinterpret_cast<std::uint8_t*>(module);
161
162 const auto size = pattern_bytes.size();
163 const auto data = pattern_bytes.data();
164
165 for (auto i = 0ul; i < size_of_image - size; ++i) {
166 bool found = true;
167 for (auto j = 0ul; j < size; ++j) {
168 if (scan_bytes[i + j] != data[j] && data[j] != 0xDEADBEEF) {
169 found = false;
170 break;
171 }
172 }
173 if (found)
174 return &scan_bytes[i];
175 }
176 return nullptr;
177}

256void ModuleScan(const ProcessInfo& context, const bool unsigned_only) {
257
258 auto [cheats] = GetLocalSignatures();
259
260 Log("\nBeginning module memory scan...\n");
261
262 // ReSharper disable once CppLocalVariableMayBeConst
263 auto dlls = EnumerateModules(context);
264
265 // For each DLL in the process
266 for (const auto& dll : dlls) {
267
268 WCHAR module_path[MAX_PATH];
269 static constinit int size = sizeof(module_path) / sizeof(TCHAR);
270
271 if (!GetModuleFileNameExW(context.handle, dll, module_path, size))
272 return;
273
274 auto scan = [&cheats, &dll, &module_path, &context] {
275
276 if (dll == context.this_module)
277 return;
278
279 // For every cheat
280 for (const auto& [cheat_name, signatures] : cheats) {
281
282 // Try every known pattern
283 for (const auto& pattern : signatures) {
284
285 // Check for pattern match
286 if (PatternScan(dll, pattern)) {
287
288 DetectionInfo detection_info{
289 .cheat_name = cheat_name,
290 .scan_type = "Module scan",
291 .cheat_path = module_path
292 };
293 Ban(detection_info);
294 }
295
296 }
297
298 }
299 };
300
301 if (unsigned_only) {
302 if (!VerifyModule(module_path)) {
303 scan();
304 Log(module_path);
305 }
306 }
307 else {
308 scan();
309 Log(module_path);
310 }
311 }
312
313 Log("\nFinished module memory scan...\n");
314}

352void FullScan(const ProcessInfo& context) {
353 const std::vector<DWORD> mask{ PAGE_EXECUTE, PAGE_EXECUTE_READWRITE, PAGE_EXECUTE_READ, PAGE_EXECUTE_WRITECOPY };
354
355 MEMORY_BASIC_INFORMATION mbi{};
356 LPVOID offset = nullptr;
357
358 while (VirtualQueryEx(GetCurrentProcess(), offset, &mbi, sizeof mbi)) {
359
360 // Move the next scan onto the next region by offsetting by the current region's size
361 const auto region = reinterpret_cast<DWORD_PTR>(mbi.BaseAddress) + mbi.RegionSize;
362 offset = reinterpret_cast<LPVOID>(region);
363
364 // Check if this memory region is executable
365 if (std::ranges::find(mask, mbi.AllocationProtect) != mask.end()) {
366
367 MemoryRegion memory_region{
368 .size = mbi.RegionSize,
369 .start_address = static_cast<std::uint8_t*>(mbi.BaseAddress)
370 };
371
372 RegionScan(context, memory_region);
373 }
374 }
