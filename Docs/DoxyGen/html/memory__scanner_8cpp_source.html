46bool VerifyModule(const LPCWSTR source_file) {
47 WINTRUST_FILE_INFO file_data{};
48 file_data.cbStruct = sizeof(WINTRUST_FILE_INFO);
49 file_data.pcwszFilePath = source_file;
50
51 GUID policy_guid = WINTRUST_ACTION_GENERIC_VERIFY_V2;
52 WINTRUST_DATA win_trust_data{
53 .cbStruct = sizeof(win_trust_data),
54 .pPolicyCallbackData = nullptr, // Use default code signing EKU
55 .pSIPClientData = nullptr, // Not applicable
56 .dwUIChoice = WTD_UI_NONE, // Disable WVT UI
57 .fdwRevocationChecks = WTD_REVOKE_NONE, // No revocation checking.
58 .dwUnionChoice = WTD_CHOICE_FILE, // Verify an embedded signature on a file
59 .pFile = &file_data, // Setup pointer to file data
60 .dwStateAction = WTD_STATEACTION_VERIFY, // Verify action
61 .hWVTStateData = nullptr, // Verification sets this value
62 .dwUIContext = 0 // Not applicable with no UI
63 };
64
65 bool verified = false;
66
67 // WinVerifyTrust verifies signatures as specified by the GUID and win_trust_data
68 if (WinVerifyTrust(nullptr, &policy_guid, &win_trust_data) == ERROR_SUCCESS) {
69 verified = true;
70 }
71
72 // Any hWVTStateData must be released by a call with close.
73 win_trust_data.dwStateAction = WTD_STATEACTION_CLOSE;
74 return verified;
75}

84std::vector<unsigned long> PatternToByte(const std::string& signature) {
85 // Pattern of bytes to be returned
86 std::vector<unsigned long> bytes{};
87
88 // Pointer to the start of the signature
89 const auto c_signature = const_cast<char*>(signature.c_str());
90
91 // Start and end of pattern
92 const auto start = c_signature;
93 const auto end = c_signature + signature.length();
94
95 // Loop over pattern
96 for (char* byte = start; byte < end; ++byte) {
97
98 // Check if signature has wildcard at this position
99 if (*byte == '*') {
100 ++byte;
101 if (*byte == '*')
102 ++byte;
103
104 // Flag value pushed back
105 bytes.push_back(0xDEADBEEF);
106 }
107 else {
108 // Store as hex
109 bytes.push_back(std::strtoul(byte, &byte, 16));
110 }
111 }
112
113 return bytes;
114}

126std::uint8_t* PatternScan(const HMODULE module, const std::string& signature) {
127 const auto dos_header = reinterpret_cast<PIMAGE_DOS_HEADER>(module);
128 const auto e_lfanew = reinterpret_cast<std::uint8_t*>(module) + dos_header->e_lfanew;
129 const auto nt_headers = reinterpret_cast<PIMAGE_NT_HEADERS>(e_lfanew);
130
131 const auto size_of_image = nt_headers->OptionalHeader.SizeOfImage;
132 const auto pattern_bytes = PatternToByte(signature);
133 const auto scan_bytes = reinterpret_cast<std::uint8_t*>(module);
134
135 const auto size = pattern_bytes.size();
136 const auto data = pattern_bytes.data();
137
138 for (auto i = 0ul; i < size_of_image - size; ++i) {
139 bool found = true;
140 for (auto j = 0ul; j < size; ++j) {
141 if (scan_bytes[i + j] != data[j] && data[j] != 0xDEADBEEF) {
142 found = false;
143 break;
144 }
145 }
146 if (found)
147 return &scan_bytes[i];
148 }
149 return nullptr;
150}

204void ModuleScan(const ProcessInfo& context, const bool unsigned_only) {
205
206 auto [cheats] = GetSignatures(context);
207
208 Log("\nBeginning module memory scan...\n");
209
210 // ReSharper disable once CppLocalVariableMayBeConst
211 auto dlls = EnumerateModules(context);
212
213 // For each DLL in the process
214 for (const auto& dll : dlls) {
215
216 WCHAR module_path[MAX_PATH];
217 static constinit int size = sizeof(module_path) / sizeof(TCHAR);
218
219 if (!GetModuleFileNameExW(context.handle, dll, module_path, size))
220 return;
221
222 auto scan = [&cheats, &dll, &module_path, &context] {
223
224 if (dll == context.this_module)
225 return;
226
227 // For every cheat
228 for (const auto& [cheat_name, signatures] : cheats) {
229
230 // Try every known pattern
231 for (const auto& pattern : signatures) {
232
233 // Check for pattern match
234 if (PatternScan(dll, pattern)) {
235
236 DetectionInfo detection_info{
237 .cheat_name = cheat_name,
238 .scan_type = "Module scan",
239 .cheat_path = module_path
240 };
241 Ban(detection_info);
242 }
243
244 }
245
246 }
247 };
248
249 if (unsigned_only) {
250 if (!VerifyModule(module_path)) {
251 scan();
252 Log(module_path);
253 }
254 }
255 else {
256 scan();
257 Log(module_path);
258 }
259 }

261 Log("\nFinished module memory scan...\n");
262}

300void FullScan(const ProcessInfo& context) {
301 const std::vector<DWORD> mask{ PAGE_EXECUTE, PAGE_EXECUTE_READWRITE, PAGE_EXECUTE_READ, PAGE_EXECUTE_WRITECOPY };
302
303 MEMORY_BASIC_INFORMATION mbi{};
304 LPVOID offset = nullptr;
305
306 while (VirtualQueryEx(GetCurrentProcess(), offset, &mbi, sizeof mbi)) {
307
308 // Move the next scan onto the next region by offsetting by the current region's size
309 const auto region = reinterpret_cast<DWORD_PTR>(mbi.BaseAddress) + mbi.RegionSize;
310 offset = reinterpret_cast<LPVOID>(region);
311
312 // Check if this memory region is executable
313 if (std::ranges::find(mask, mbi.AllocationProtect) != mask.end()) {
314
315 MemoryRegion memory_region{
316 .size = mbi.RegionSize,
317 .start_address = static_cast<std::uint8_t*>(mbi.BaseAddress)
318 };
319
320 RegionScan(context, memory_region);
321 }
322 }
