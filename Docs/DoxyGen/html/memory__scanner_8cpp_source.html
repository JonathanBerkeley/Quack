36// Verifies signature for source file, based on MSDN example code
37// https://docs.microsoft.com/en-us/windows/win32/seccrypto/example-c-program--verifying-the-signature-of-a-pe-file
38bool VerifyModule(const LPCWSTR source_file) {
39 WINTRUST_FILE_INFO file_data{};
40 file_data.cbStruct = sizeof(WINTRUST_FILE_INFO);
41 file_data.pcwszFilePath = source_file;
42
43 GUID policy_guid = WINTRUST_ACTION_GENERIC_VERIFY_V2;
44 WINTRUST_DATA win_trust_data{
45 .cbStruct = sizeof(win_trust_data),
46 .pPolicyCallbackData = nullptr, // Use default code signing EKU
47 .pSIPClientData = nullptr, // Not applicable
48 .dwUIChoice = WTD_UI_NONE, // Disable WVT UI
49 .fdwRevocationChecks = WTD_REVOKE_NONE, // No revocation checking.
50 .dwUnionChoice = WTD_CHOICE_FILE, // Verify an embedded signature on a file
51 .pFile = &file_data, // Setup pointer to file data
52 .dwStateAction = WTD_STATEACTION_VERIFY, // Verify action
53 .hWVTStateData = nullptr, // Verification sets this value
54 .dwUIContext = 0 // Not applicable with no UI
55 };
56
57 bool verified = false;
58
59 // WinVerifyTrust verifies signatures as specified by the GUID and win_trust_data
60 if (WinVerifyTrust(nullptr, &policy_guid, &win_trust_data) == ERROR_SUCCESS) {
61 verified = true;
62 }
63
64 // Any hWVTStateData must be released by a call with close.
65 win_trust_data.dwStateAction = WTD_STATEACTION_CLOSE;
66 return verified;
67}

76std::vector<unsigned long> PatternToByte(const std::string& signature) {
77 // Pattern of bytes to be returned
78 std::vector<unsigned long> bytes{};
79
80 // Pointer to the start of the signature
81 const auto c_signature = const_cast<char*>(signature.c_str());
82
83 // Start and end of pattern
84 const auto start = c_signature;
85 const auto end = c_signature + signature.length();
86
87 // Loop over pattern
88 for (char* byte = start; byte < end; ++byte) {
89
90 // Check if signature has wildcard at this position
91 if (*byte == '*') {
92 ++byte;
93 if (*byte == '*')
94 ++byte;
95
96 // Flag value pushed back
97 bytes.push_back(0xDEADBEEF);
98 }
99 else {
100 // Store as hex
101 bytes.push_back(std::strtoul(byte, &byte, 16));
102 }
103 }
104
105 return bytes;
106}

118std::uint8_t* PatternScan(const HMODULE module, const std::string& signature) {
119 const auto dos_header = reinterpret_cast<PIMAGE_DOS_HEADER>(module);
120 const auto e_lfanew = reinterpret_cast<std::uint8_t*>(module) + dos_header->e_lfanew;
121 const auto nt_headers = reinterpret_cast<PIMAGE_NT_HEADERS>(e_lfanew);
122
123 const auto size_of_image = nt_headers->OptionalHeader.SizeOfImage;
124 const auto pattern_bytes = PatternToByte(signature);
125 const auto scan_bytes = reinterpret_cast<std::uint8_t*>(module);
126
127 const auto size = pattern_bytes.size();
128 const auto data = pattern_bytes.data();
129
130 for (auto i = 0ul; i < size_of_image - size; ++i) {
131 bool found = true;
132 for (auto j = 0ul; j < size; ++j) {
133 if (scan_bytes[i + j] != data[j] && data[j] != 0xDEADBEEF) {
134 found = false;
135 break;
136 }
137 }
138 if (found)
139 return &scan_bytes[i];
140 }
141 return nullptr;
142}

181

184 Log("\nBeginning module memory scan...\n");

188
189 // For each DLL in the process

191
192 WCHAR module_path[MAX_PATH];
193 static constinit int size = sizeof(module_path) / sizeof(TCHAR);
194
195 if (!GetModuleFileNameExW(context.handle, dll, module_path, size))
196 return;
197
198 auto scan = [&cheats, &dll, &module_path, &context] {
199
200 if (dll == context.this_module)
201 return;
202
203 // For every cheat

205
206 // Try every known pattern
207 for (const auto& pattern : signatures) {
208
209 // Check for pattern match

211
212 DetectionInfo detection_info{
213 .cheat_name = cheat_name,
214 .scan_type = "Module scan",

217 Ban(detection_info);

267void FullScan(const ProcessInfo& context) {
268 const std::vector<DWORD> mask{ PAGE_EXECUTE, PAGE_EXECUTE_READWRITE, PAGE_EXECUTE_READ, PAGE_EXECUTE_WRITECOPY };
269
270 MEMORY_BASIC_INFORMATION mbi{};
271 LPVOID offset = nullptr;
272
273 while (VirtualQueryEx(GetCurrentProcess(), offset, &mbi, sizeof mbi)) {
274
275 // Move the next scan onto the next region by offsetting by the current region's size
276 const auto region = reinterpret_cast<DWORD_PTR>(mbi.BaseAddress) + mbi.RegionSize;
277 offset = reinterpret_cast<LPVOID>(region);
278
279 // Check if this memory region is executable
280 if (std::ranges::find(mask, mbi.AllocationProtect) != mask.end()) {
281
282 MemoryRegion memory_region{
283 .size = mbi.RegionSize,
284 .start_address = static_cast<std::uint8_t*>(mbi.BaseAddress)
285 };
286
287 RegionScan(context, memory_region);
288 }
289 }
